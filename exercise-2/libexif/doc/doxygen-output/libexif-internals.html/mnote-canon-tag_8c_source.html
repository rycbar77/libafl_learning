130 mnote_canon_tag_get_name_sub (MnoteCanonTag t, unsigned int s, ExifDataOption o)
131 {
132  unsigned int i;
133  int tag_found = 0;
134 
135  for (i = 0; i < sizeof (table_sub) / sizeof (table_sub[0]); i++) {
136  if (table_sub[i].tag == t) {
137  if (table_sub[i].subtag == s)
138  return table_sub[i].name;
139  tag_found = 1;
140  }
141  }
142  if (!tag_found || !(o & EXIF_DATA_OPTION_IGNORE_UNKNOWN_TAGS))
143  return mnote_canon_tag_get_name (t);
144  else
145  return NULL;
146 }
