100 void
101 exif_loader_write_file (ExifLoader *l, const char *path)
102 {
103  FILE *f;
104  int size;
105  unsigned char data[1024];
106 
107  if (!l)
108  return;
109 
110  f = fopen (path, "rb");
111  if (!f) {
112  exif_log (l->log, EXIF_LOG_CODE_NONE, "ExifLoader",
113  _("The file '%s' could not be opened."), path);
114  return;
115  }
116  while (1) {
117  size = fread (data, 1, sizeof (data), f);
118  if (size <= 0)
119  break;
120  if (!exif_loader_write (l, data, size))
121  break;
122  }
123  fclose (f);
124 }

126 static unsigned int
127 exif_loader_copy (ExifLoader *eld, unsigned char *buf, unsigned int len)
128 {
129  if (!eld || (len && !buf) || (eld->bytes_read >= eld->size))
130  return 0;
131 
132  /* If needed, allocate the buffer. */
133  if (!eld->buf)
134  eld->buf = exif_loader_alloc (eld, eld->size);
135  if (!eld->buf)
136  return 0;
137 
138  /* Copy memory */
139  len = MIN (len, eld->size - eld->bytes_read);
140  memcpy (eld->buf + eld->bytes_read, buf, len);
141  eld->bytes_read += len;
142 
143  return (eld->bytes_read >= eld->size) ? 0 : 1;
144 }

146 unsigned char
147 exif_loader_write (ExifLoader *eld, unsigned char *buf, unsigned int len)
148 {
149  unsigned int i;
150 
151  if (!eld || (len && !buf))
152  return 0;
153 
154  switch (eld->state) {
155  case EL_EXIF_FOUND:
156  return exif_loader_copy (eld, buf, len);
157  case EL_SKIP_BYTES:
158  if (eld->size > len) {
159  eld->size -= len;
160  return 1;
161  }
162  len -= eld->size;
163  buf += eld->size;
164  eld->size = 0;
165  eld->b_len = 0;
166  switch (eld->data_format) {
167  case EL_DATA_FORMAT_FUJI_RAW:
168  eld->state = EL_READ_SIZE_BYTE_24;
169  break;
170  default:
171  eld->state = EL_READ;
172  break;
173  }
174  break;
175  default:
176  break;
177  }
178 
179  exif_log (eld->log, EXIF_LOG_CODE_DEBUG, "ExifLoader",
180  "Scanning %i byte(s) of data...", len);
181 
182  /*
183  * First fill the small buffer. Only continue if the buffer
184  * is filled. Note that EXIF data contains at least 12 bytes.
185  */
186  i = MIN (len, sizeof (eld->b) - eld->b_len);
187  if (i) {
188  memcpy (&eld->b[eld->b_len], buf, i);
189  eld->b_len += i;
190  if (eld->b_len < sizeof (eld->b))
191  return 1;
192  buf += i;
193  len -= i;
194  }
195 
196  switch (eld->data_format) {
197  case EL_DATA_FORMAT_UNKNOWN:
198 
199  /* Check the small buffer against known formats. */
200  if (!memcmp (eld->b, "FUJIFILM", 8)) {
201 
202  /* Skip to byte 84. There is another offset there. */
203  eld->data_format = EL_DATA_FORMAT_FUJI_RAW;
204  eld->size = 84;
205  eld->state = EL_SKIP_BYTES;
206  eld->size = 84;
207 
208  } else if (!memcmp (eld->b + 2, ExifHeader, sizeof (ExifHeader))) {
209 
210  /* Read the size (2 bytes). */
211  eld->data_format = EL_DATA_FORMAT_EXIF;
212  eld->state = EL_READ_SIZE_BYTE_08;
213  }
214  default:
215  break;
216  }
217 
218  for (i = 0; i < sizeof (eld->b); i++)
219  switch (eld->state) {
220  case EL_EXIF_FOUND:
221  if (!exif_loader_copy (eld, eld->b + i,
222  sizeof (eld->b) - i))
223  return 0;
224  return exif_loader_copy (eld, buf, len);
225  case EL_SKIP_BYTES:
226  eld->size--;
227  if (!eld->size)
228  eld->state = EL_READ;
229  break;
230 
231  case EL_READ_SIZE_BYTE_24:
232  eld->size |= eld->b[i] << 24;
233  eld->state = EL_READ_SIZE_BYTE_16;
234  break;
235  case EL_READ_SIZE_BYTE_16:
236  eld->size |= eld->b[i] << 16;
237  eld->state = EL_READ_SIZE_BYTE_08;
238  break;
239  case EL_READ_SIZE_BYTE_08:
240  eld->size |= eld->b[i] << 8;
241  eld->state = EL_READ_SIZE_BYTE_00;
242  break;
243  case EL_READ_SIZE_BYTE_00:
244  eld->size |= eld->b[i] << 0;
245  switch (eld->data_format) {
246  case EL_DATA_FORMAT_JPEG:
247  eld->state = EL_SKIP_BYTES;
248  eld->size -= 2;
249  break;
250  case EL_DATA_FORMAT_FUJI_RAW:
251  eld->data_format = EL_DATA_FORMAT_EXIF;
252  eld->state = EL_SKIP_BYTES;
253  eld->size -= 86;
254  break;
255  case EL_DATA_FORMAT_EXIF:
256  eld->state = EL_EXIF_FOUND;
257  break;
258  default:
259  break;
260  }
261  break;
262 
263  default:
264  switch (eld->b[i]) {
265  case JPEG_MARKER_APP1:
266  if (!memcmp (eld->b + i + 3, ExifHeader, MIN(sizeof (ExifHeader), MAX(0, sizeof (eld->b) - i - 3)))) {
267  eld->data_format = EL_DATA_FORMAT_EXIF;
268  } else {
269  eld->data_format = EL_DATA_FORMAT_JPEG; /* Probably JFIF - keep searching for APP1 EXIF*/
270  }
271  eld->size = 0;
272  eld->state = EL_READ_SIZE_BYTE_08;
273  break;
274  case JPEG_MARKER_APP0:
275  case JPEG_MARKER_APP2:
276  case JPEG_MARKER_APP13:
277  case JPEG_MARKER_COM:
278  eld->data_format = EL_DATA_FORMAT_JPEG;
279  eld->size = 0;
280  eld->state = EL_READ_SIZE_BYTE_08;
281  break;
282  case 0xff:
283  case JPEG_MARKER_SOI:
284  break;
285  default:
286  exif_log (eld->log,
287  EXIF_LOG_CODE_CORRUPT_DATA,
288  "ExifLoader", _("The data supplied "
289  "does not seem to contain "
290  "EXIF data."));
291  exif_loader_reset (eld);
292  return 0;
293  }
294  }
295 
296  /*
297  * If we reach this point, the buffer has not been big enough
298  * to read all data we need. Fill it with new data.
299  */
300  eld->b_len = 0;
301  return exif_loader_write (eld, buf, len);
302 }
