84 static void
85 exif_mnote_data_olympus_save (ExifMnoteData *ne,
86  unsigned char **buf, unsigned int *buf_size)
87 {
88  ExifMnoteDataOlympus *n = (ExifMnoteDataOlympus *) ne;
89  unsigned int i, o, s, doff, base = 0, o2 = 6 + 2;
90  int datao = 0;
91 
92  if (!n || !buf || !buf_size) return;
93 
94  /*
95  * Allocate enough memory for all entries and the number of entries.
96  */
97  *buf_size = 6 + 2 + 2 + n->count * 12;
98  switch (n->version) {
99  case olympusV1:
100  *buf = exif_mem_alloc (ne->mem, *buf_size);
101  if (!*buf) return;
102 
103  /* Write the header and the number of entries. */
104  strcpy ((char *)*buf, "OLYMP");
105  exif_set_short (*buf + 6, n->order, (ExifShort) 1);
106  datao = n->offset;
107  break;
108  case olympusV2:
109  *buf_size += 8-6 + 4;
110  *buf = exif_mem_alloc (ne->mem, *buf_size);
111  if (!*buf) return;
112 
113  /* Write the header and the number of entries. */
114  strcpy ((char *)*buf, "OLYMPUS");
115  exif_set_short (*buf + 8, n->order, (ExifShort) (
116  (n->order == EXIF_BYTE_ORDER_INTEL) ?
117  ('I' << 8) | 'I' :
118  ('M' << 8) | 'M'));
119  exif_set_short (*buf + 10, n->order, (ExifShort) 3);
120  o2 += 4;
121  break;
122  case nikonV1:
123  base = MNOTE_NIKON1_TAG_BASE;
124 
125  /* v1 has offsets based to main IFD, not makernote IFD */
126  datao += n->offset + 10;
127  /* subtract the size here, so the increment in the next case will not harm us */
128  *buf_size -= 8 + 2;
129  /* Fall through */
130  case nikonV2:
131  *buf_size += 8 + 2;
132  *buf_size += 4; /* Next IFD pointer */
133  *buf = exif_mem_alloc (ne->mem, *buf_size);
134  if (!*buf) return;
135 
136  /* Write the header and the number of entries. */
137  strcpy ((char *)*buf, "Nikon");
138  (*buf)[6] = n->version;
139 
140  if (n->version == nikonV2) {
141  exif_set_short (*buf + 10, n->order, (ExifShort) (
142  (n->order == EXIF_BYTE_ORDER_INTEL) ?
143  ('I' << 8) | 'I' :
144  ('M' << 8) | 'M'));
145  exif_set_short (*buf + 12, n->order, (ExifShort) 0x2A);
146  exif_set_long (*buf + 14, n->order, (ExifShort) 8);
147  o2 += 2 + 8;
148  }
149  datao -= 10;
150  /* Reset next IFD pointer */
151  exif_set_long (*buf + o2 + 2 + n->count * 12, n->order, 0);
152  break;
153 
154  default:
155  return;
156  }
157 
158  exif_set_short (*buf + o2, n->order, (ExifShort) n->count);
159  o2 += 2;
160 
161  /* Save each entry */
162  for (i = 0; i < n->count; i++) {
163  o = o2 + i * 12;
164  exif_set_short (*buf + o + 0, n->order,
165  (ExifShort) (n->entries[i].tag - base));
166  exif_set_short (*buf + o + 2, n->order,
167  (ExifShort) n->entries[i].format);
168  exif_set_long (*buf + o + 4, n->order,
169  n->entries[i].components);
170  o += 8;
171  s = exif_format_get_size (n->entries[i].format) *
172  n->entries[i].components;
173  if (s > 4) {
174  doff = *buf_size;
175  *buf_size += s;
176  *buf = exif_mem_realloc (ne->mem, *buf,
177  sizeof (char) * *buf_size);
178  if (!*buf) return;
179  exif_set_long (*buf + o, n->order, datao + doff);
180  } else
181  doff = o;
182 
183  /* Write the data. */
184  if (n->entries[i].data) {
185  memcpy (*buf + doff, n->entries[i].data, s);
186  } else {
187  /* Most certainly damaged input file */
188  memset (*buf + doff, 0, s);
189  }
190  }

193 static void
194 exif_mnote_data_olympus_load (ExifMnoteData *en,
195  const unsigned char *buf, unsigned int buf_size)
196 {
197  ExifMnoteDataOlympus *n = (ExifMnoteDataOlympus *) en;
198  ExifShort c;
199  unsigned int i, s, o, o2 = 0, datao = 6, base = 0;
200 
201  if (!n || !buf) return;
202 
203  /* Start of interesting data */
204  o2 = 6 + n->offset;
205 
206  /*
207  * Olympus headers start with "OLYMP" and need to have at least
208  * a size of 22 bytes (6 for 'OLYMP', 2 other bytes, 2 for the
209  * number of entries, and 12 for one entry.
210  *
211  * Nikon headers start with "Nikon" (6 bytes including '\0'),
212  * version number (1 or 2).
213  *
214  * Version 1 continues with 0, 1, 0, number_of_tags,
215  * or just with number_of_tags (models D1H, D1X...).
216  *
217  * Version 2 continues with an unknown byte (0 or 10),
218  * two unknown bytes (0), "MM" or "II", another byte 0 and
219  * lastly 0x2A.
220  */
221  if (buf_size - n->offset < 22) return;
222  if (!memcmp (buf + o2, "OLYMP", 6)) {
223  exif_log (en->log, EXIF_LOG_CODE_DEBUG, "ExifMnoteDataOlympus",
224  "Parsing Olympus maker note v1...");
225 
226  /* The number of entries is at position 8. */
227  n->version = olympusV1;
228  if (buf[o2 + 6] == 1)
229  n->order = EXIF_BYTE_ORDER_INTEL;
230  else if (buf[o2 + 6 + 1] == 1)
231  n->order = EXIF_BYTE_ORDER_MOTOROLA;
232  o2 += 8;
233 
234  } else if (!memcmp (buf + o2, "OLYMPUS", 8)) {
235  /* Olympus S760, S770 */
236  datao = o2;
237  o2 += 8;
238  exif_log (en->log, EXIF_LOG_CODE_DEBUG, "ExifMnoteDataOlympus",
239  "Parsing Olympus maker note v2 (0x%02x, %02x, %02x, %02x)...",
240  buf[o2], buf[o2 + 1], buf[o2 + 2], buf[o2 + 3]);
241 
242  if ((buf[o2] == 'I') && (buf[o2 + 1] == 'I'))
243  n->order = EXIF_BYTE_ORDER_INTEL;
244  else if ((buf[o2] == 'M') && (buf[o2 + 1] == 'M'))
245  n->order = EXIF_BYTE_ORDER_MOTOROLA;
246 
247  /* The number of entries is at position 8+4. */
248  n->version = olympusV2;
249  o2 += 4;
250 
251  } else if (!memcmp (buf + o2, "Nikon", 6)) {
252  o2 += 6;
253  exif_log (en->log, EXIF_LOG_CODE_DEBUG, "ExifMnoteDataOlympus",
254  "Parsing Nikon maker note (0x%02x, %02x, %02x, "
255  "%02x, %02x, %02x, %02x, %02x)...",
256  buf[o2 + 0], buf[o2 + 1], buf[o2 + 2], buf[o2 + 3],
257  buf[o2 + 4], buf[o2 + 5], buf[o2 + 6], buf[o2 + 7]);
258  /* The first byte is the version. */
259  if (o2 >= buf_size) return;
260  n->version = buf[o2];
261  o2 += 1;
262 
263  /* Skip an unknown byte (00 or 0A). */
264  o2 += 1;
265 
266  switch (n->version) {
267  case nikonV1:
268 
269  base = MNOTE_NIKON1_TAG_BASE;
270  break;
271 
272  case nikonV2:
273 
274  /* Skip 2 unknown bytes (00 00). */
275  o2 += 2;
276 
277  /*
278  * Byte order. From here the data offset
279  * gets calculated.
280  */
281  datao = o2;
282  if (o2 >= buf_size) return;
283  if (!strncmp ((char *)&buf[o2], "II", 2))
284  n->order = EXIF_BYTE_ORDER_INTEL;
285  else if (!strncmp ((char *)&buf[o2], "MM", 2))
286  n->order = EXIF_BYTE_ORDER_MOTOROLA;
287  else {
288  exif_log (en->log, EXIF_LOG_CODE_DEBUG,
289  "ExifMnoteDatalympus", "Unknown "
290  "byte order '%c%c'", buf[o2],
291  buf[o2 + 1]);
292  return;
293  }
294  o2 += 2;
295 
296  /* Skip 2 unknown bytes (00 2A). */
297  o2 += 2;
298 
299  /* Go to where the number of entries is. */
300  if (o2 >= buf_size) return;
301  o2 = datao + exif_get_long (buf + o2, n->order);
302  break;
303 
304  default:
305  exif_log (en->log, EXIF_LOG_CODE_DEBUG,
306  "ExifMnoteDataOlympus", "Unknown version "
307  "number %i.", n->version);
308  return;
309  }
310  } else if (!memcmp (buf + o2, "\0\x1b", 2)) {
311  n->version = nikonV2;
312  } else {
313  return;
314  }
315 
316  /* Number of entries */
317  if (o2 >= buf_size) return;
318  c = exif_get_short (buf + o2, n->order);
319  o2 += 2;
320 
321  /* Read the number of entries and remove old ones. */
322  exif_mnote_data_olympus_clear (n);
323 
324  n->entries = exif_mem_alloc (en->mem, sizeof (MnoteOlympusEntry) * c);
325  if (!n->entries) return;
326 
327  /* Parse the entries */
328  for (i = 0; i < c; i++) {
329  o = o2 + 12 * i;
330  if (o + 12 > buf_size) return;
331 
332  n->count = i + 1;
333  n->entries[i].tag = exif_get_short (buf + o, n->order) + base;
334  n->entries[i].format = exif_get_short (buf + o + 2, n->order);
335  n->entries[i].components = exif_get_long (buf + o + 4, n->order);
336  n->entries[i].order = n->order;
337 
338  exif_log (en->log, EXIF_LOG_CODE_DEBUG, "ExifMnoteOlympus",
339  "Loading entry 0x%x ('%s')...", n->entries[i].tag,
340  mnote_olympus_tag_get_name (n->entries[i].tag));
341 
342  /*
343  * Size? If bigger than 4 bytes, the actual data is not
344  * in the entry but somewhere else (offset).
345  */
346  s = exif_format_get_size (n->entries[i].format) *
347  n->entries[i].components;
348  if (!s) continue;
349  o += 8;
350  if (s > 4) o = exif_get_long (buf + o, n->order) + datao;
351  if (o + s > buf_size) continue;
352 
353  /* Sanity check */
354  n->entries[i].data = exif_mem_alloc (en->mem, s);
355  if (!n->entries[i].data) continue;
356  n->entries[i].size = s;
357  memcpy (n->entries[i].data, buf + o, s);
358  }
359 }
