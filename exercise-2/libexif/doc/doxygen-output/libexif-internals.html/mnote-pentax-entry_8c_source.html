119 char *
120 mnote_pentax_entry_get_value (MnotePentaxEntry *entry,
121  char *val, unsigned int maxlen)
122 {
123  ExifLong vl;
124  ExifShort vs;
125  int i = 0, j = 0;
126 
127  if (!entry) return (NULL);
128 
129  memset (val, 0, maxlen);
130  maxlen--;
131 
132  switch (entry->tag) {
133  case MNOTE_PENTAX_TAG_MODE:
134  case MNOTE_PENTAX_TAG_QUALITY:
135  case MNOTE_PENTAX_TAG_FOCUS:
136  case MNOTE_PENTAX_TAG_FLASH:
137  case MNOTE_PENTAX_TAG_WHITE_BALANCE:
138  case MNOTE_PENTAX_TAG_SHARPNESS:
139  case MNOTE_PENTAX_TAG_CONTRAST:
140  case MNOTE_PENTAX_TAG_SATURATION:
141  case MNOTE_PENTAX_TAG_ISO_SPEED:
142  case MNOTE_PENTAX_TAG_COLOR:
143  CF (entry->format, EXIF_FORMAT_SHORT, val, maxlen);
144  CC (entry->components, 1, val, maxlen);
145  vs = exif_get_short (entry->data, entry->order);
146 
147  /* search the tag */
148  for (i = 0; (items[i].tag && items[i].tag != entry->tag); i++);
149  if (!items[i].tag) {
150  strncpy (val, "Internal error", maxlen);
151  break;
152  }
153 
154  /* find the value */
155  for (j = 0; items[i].elem[j].string &&
156  (items[i].elem[j].index < vs); j++);
157  if (items[i].elem[j].index != vs) {
158  snprintf (val, maxlen,
159  "Internal error (unknown value %i)", vs);
160  break;
161  }
162  snprintf (val, maxlen, "%s", items[i].elem[j].string);
163  break;
164 
165  case MNOTE_PENTAX_TAG_ZOOM:
166  CF (entry->format, EXIF_FORMAT_LONG, val, maxlen);
167  CC (entry->components, 1, val, maxlen);
168  vl = exif_get_long (entry->data, entry->order);
169  snprintf (val, maxlen, "%li", (long int) vl);
170  break;
171  case MNOTE_PENTAX_TAG_PRINTIM:
172  CF (entry->format, EXIF_FORMAT_UNDEFINED, val, maxlen);
173  CC (entry->components, 124, val, maxlen);
174  snprintf (val, maxlen, "%li bytes unknown data",
175  entry->components);
176  break;
177  case MNOTE_PENTAX_TAG_TZ_CITY:
178  CF (entry->format, EXIF_FORMAT_UNDEFINED, val, maxlen);
179  CC (entry->components, 4, val, maxlen);
180  snprintf (val, entry->components, "%s", entry->data);
181  break;
182  case MNOTE_PENTAX_TAG_TZ_DST:
183  CF (entry->format, EXIF_FORMAT_UNDEFINED, val, maxlen);
184  CC (entry->components, 4, val, maxlen);
185  snprintf (val, entry->components, "%s", entry->data);
186  break;
187  default:
188  switch (entry->format) {
189  case EXIF_FORMAT_ASCII:
190  strncpy (val, (char *)entry->data, MIN(maxlen, entry->components));
191  break;
192  case EXIF_FORMAT_SHORT:
193  vs = exif_get_short (entry->data, entry->order);
194  snprintf (val, maxlen, "%i", vs);
195  break;
196  case EXIF_FORMAT_LONG:
197  vl = exif_get_long (entry->data, entry->order);
198  snprintf (val, maxlen, "%li", (long int) vl);
199  break;
200  case EXIF_FORMAT_UNDEFINED:
201  default:
202  snprintf (val, maxlen, "%li bytes unknown data",
203  entry->components);
204  break;
205  }
206  break;
207  }
208 
209  return (val);
210 }
