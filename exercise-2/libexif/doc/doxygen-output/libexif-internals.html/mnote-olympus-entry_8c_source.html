33 #define CF(format,target,v,maxlen) \
34 { \
35  if (format != target) { \
36  snprintf (v, maxlen, \
37  _("Invalid format '%s', " \
38  "expected '%s'."), \
39  exif_format_get_name (format), \
40  exif_format_get_name (target)); \
41  break; \
42  } \
43 }
44 
45 #define CC(number,target,v,maxlen) \
46 { \
47  if (number != target) { \
48  snprintf (v, maxlen, \
49  _("Invalid number of components (%i, " \
50  "expected %i)."), (int) number, (int) target); \
51  break; \
52  } \
53 }
54 
55 #define CC2(number,t1,t2,v,maxlen) \
56 { \
57  if ((number < t1) || (number > t2)) { \
58  snprintf (v, maxlen, \
59  _("Invalid number of components (%i, " \
60  "expected %i or %i)."), (int) number, \
61  (int) t1, (int) t2); \
62  break; \
63  } \
64 }

167 char *
168 mnote_olympus_entry_get_value (MnoteOlympusEntry *entry, char *v, unsigned int maxlen)
169 {
170  char buf[30];
171  ExifLong vl;
172  ExifShort vs = 0;
173  ExifRational vr, vr2;
174  int i, j;
175  double r, b;
176 
177  if (!entry)
178  return (NULL);
179 
180  memset (v, 0, maxlen);
181  maxlen--;
182 
183  if ((!entry->data) && (entry->components > 0)) return (v);
184 
185  switch (entry->tag) {
186 
187  /* Nikon */
188  case MNOTE_NIKON_TAG_FIRMWARE:
189  CF (entry->format, EXIF_FORMAT_UNDEFINED, v, maxlen);
190  CC (entry->components, 4, v, maxlen);
191  vl = exif_get_long (entry->data, entry->order);
192  if ((vl & 0xF0F0F0F0) == 0x30303030) {
193  memcpy (v, entry->data, MIN (maxlen, 4));
194  } else {
195  snprintf (v, maxlen, "%04lx", (long unsigned int) vl);
196  }
197  break;
198  case MNOTE_NIKON_TAG_ISO:
199  CF (entry->format, EXIF_FORMAT_SHORT, v, maxlen);
200  CC (entry->components, 2, v, maxlen);
201  //vs = exif_get_short (entry->data, entry->order);
202  vs = exif_get_short (entry->data + 2, entry->order);
203  snprintf (v, maxlen, "ISO %hd", vs);
204  break;
205  case MNOTE_NIKON_TAG_ISO2:

207  CC (entry->components, 2, v, maxlen);
208  //vs = exif_get_short (entry->data, entry->order);

210  snprintf (v, maxlen, "ISO2 %hd", vs);
211  break;
212  case MNOTE_NIKON_TAG_QUALITY:
213  CF (entry->format, EXIF_FORMAT_ASCII, v, maxlen);
214  //CC (entry->components, 8, v, maxlen);
215  //vl = exif_get_long (entry->data , entry->order);
216  //printf("-> 0x%04x\n",entry->data);
217  //printf("-> 0x%s<\n",entry->data - 0);

219  //snprintf (v, maxlen, "%s<", ( entry->data - 9 );

221  case MNOTE_NIKON_TAG_COLORMODE:
222  case MNOTE_NIKON_TAG_COLORMODE1:
223  case MNOTE_NIKON_TAG_WHITEBALANCE:
224  case MNOTE_NIKON_TAG_SHARPENING:
225  case MNOTE_NIKON_TAG_FOCUSMODE:
226  case MNOTE_NIKON_TAG_FLASHSETTING:
227  case MNOTE_NIKON_TAG_ISOSELECTION:
228  case MNOTE_NIKON_TAG_FLASHMODE:
229  case MNOTE_NIKON_TAG_IMAGEADJUSTMENT:
230  case MNOTE_NIKON_TAG_ADAPTER:
231  case MNOTE_NIKON_TAG_SATURATION2:
232  CF (entry->format, EXIF_FORMAT_ASCII, v, maxlen);
233  memcpy(v, entry->data, MIN (maxlen, entry->components));
234  break;
235  case MNOTE_NIKON_TAG_TOTALPICTURES:
236  CF (entry->format, EXIF_FORMAT_LONG, v, maxlen);
237  CC (entry->components, 1, v, maxlen);
238  vl = exif_get_long (entry->data, entry->order);
239  snprintf (v, maxlen, "%lu", (long unsigned int) vl );
240  break;
241  case MNOTE_NIKON_TAG_LENS_FSTOPS:
242  case MNOTE_NIKON_TAG_EXPOSUREDIFF: {
243  unsigned char a,b,c,d;
244  CF (entry->format, EXIF_FORMAT_UNDEFINED, v, maxlen);
245  CC (entry->components, 4, v, maxlen);
246  vl = exif_get_long (entry->data, entry->order);
247  a = (vl>>24)&0xff; b = (vl>>16)&0xff; c = (vl>>8)&0xff; d = (vl)&0xff;
248  snprintf (v, maxlen, "%.1f", c?(float)a*((float)b/(float)c):0 );
249  break;
250  }
251  case MNOTE_NIKON_TAG_FLASHEXPCOMPENSATION:
252  case MNOTE_NIKON_TAG_FLASHEXPOSUREBRACKETVAL:
253  CF (entry->format, EXIF_FORMAT_UNDEFINED, v, maxlen);
254  CC (entry->components, 4, v, maxlen);
255  vl = exif_get_long (entry->data, entry->order);
256  snprintf (v, maxlen, "%.1f", ((long unsigned int) vl>>24)/6.0 );
257  break;
258  case MNOTE_NIKON_TAG_SATURATION:
259  case MNOTE_NIKON_TAG_WHITEBALANCEFINE:
260  CF (entry->format, EXIF_FORMAT_SSHORT, v, maxlen);
261  CC (entry->components, 1, v, maxlen);
262  vs = exif_get_short (entry->data, entry->order);
263  snprintf (v, maxlen, "%hd", vs);
264  break;
265  case MNOTE_NIKON_TAG_WHITEBALANCERB:

267  CC (entry->components, 4, v, maxlen);
268  vr = exif_get_rational (entry->data, entry->order);
269  r = (double)vr.numerator / vr.denominator;
270  vr = exif_get_rational (entry->data+8, entry->order);
271  b = (double)vr.numerator / vr.denominator;
272  //printf("numerator %li, denominator %li\n", vr.numerator, vr.denominator);
273  snprintf (v, maxlen, _("Red Correction %f, Blue Correction %f"), r,b);
274  break;
275  case MNOTE_NIKON_TAG_MANUALFOCUSDISTANCE:
276  CF (entry->format, EXIF_FORMAT_RATIONAL, v, maxlen);
277  CC (entry->components, 1, v, maxlen);
278  vr = exif_get_rational (entry->data, entry->order);
279  if (vr.numerator) {
280  r = (double)vr.numerator / vr.denominator;
281  snprintf (v, maxlen, _("%2.2f meters"), r);
282  } else {
283  strncpy (v, _("No manual focus selection"), maxlen);
284  }
285  break;
286  case MNOTE_NIKON_TAG_DIGITALZOOM:
287  case MNOTE_NIKON1_TAG_DIGITALZOOM:
288  CF (entry->format, EXIF_FORMAT_RATIONAL, v, maxlen);

290  vr = exif_get_rational (entry->data, entry->order);

292  snprintf (v, maxlen, "%2.2f", r);
293  break;

299  r = (double)vr.numerator / vr.denominator;
300  b = (double)vr2.numerator / vr2.denominator;

303  case MNOTE_NIKON_TAG_HUE:
304  CF (entry->format, EXIF_FORMAT_SSHORT, v, maxlen);

307  snprintf (v, maxlen, "%hd", vs);
308  break;
309  case MNOTE_NIKON_TAG_AFFOCUSPOSITION:
310  CF (entry->format, EXIF_FORMAT_UNDEFINED, v, maxlen);
311  CC (entry->components, 4, v, maxlen);
312  switch ( *( entry->data+1) ) {
313  case 0: strncpy (v, _("AF Position: Center"), maxlen); break;
314  case 1: strncpy (v, _("AF Position: Top"), maxlen); break;
315  case 2: strncpy (v, _("AF Position: Bottom"), maxlen); break;
316  case 3: strncpy (v, _("AF Position: Left"), maxlen); break;
317  case 4: strncpy (v, _("AF Position: Right"), maxlen); break;
318  case 5: strncpy (v, _("AF Position: Upper-left"), maxlen); break;
319  case 6: strncpy (v, _("AF Position: Upper-right"), maxlen); break;

327  case MNOTE_OLYMPUS_TAG_DIGIZOOM:
328  if (entry->format == EXIF_FORMAT_RATIONAL) {
329  CC (entry->components, 1, v, maxlen);
330  vr = exif_get_rational (entry->data, entry->order);
331  r = (double)vr.numerator / vr.denominator;
332  if (!vr.numerator) {
333  strncpy (v, _("None"), maxlen);
334  } else {
335  snprintf (v, maxlen, "%2.2f", r);
336  }
337  break;
338  }
339  /* fall through to handle SHORT version of this tag */
340  case MNOTE_NIKON_TAG_LENSTYPE:
341  case MNOTE_NIKON_TAG_FLASHUSED:
342  case MNOTE_NIKON1_TAG_QUALITY:
343  case MNOTE_NIKON1_TAG_COLORMODE:
344  case MNOTE_NIKON1_TAG_IMAGEADJUSTMENT:
345  case MNOTE_NIKON1_TAG_CCDSENSITIVITY:
346  case MNOTE_NIKON1_TAG_WHITEBALANCE:
347  case MNOTE_NIKON1_TAG_CONVERTER:
348  case MNOTE_OLYMPUS_TAG_QUALITY:
349  case MNOTE_OLYMPUS_TAG_MACRO:
350  case MNOTE_OLYMPUS_TAG_FLASHMODE:
351  case MNOTE_OLYMPUS_TAG_SHARPNESS:
352  case MNOTE_OLYMPUS_TAG_CONTRAST:
353  case MNOTE_OLYMPUS_TAG_MANFOCUS:
354  /* search the tag */
355  for (i = 0; (items[i].tag && items[i].tag != entry->tag); i++);
356  if (!items[i].tag) {
357  strncpy (v, _("Internal error"), maxlen);
358  break;
359  }
360  CF (entry->format, items[i].fmt, v, maxlen);
361  CC (entry->components, 1, v, maxlen);
362  switch (entry->format) {
363  case EXIF_FORMAT_BYTE:
364  case EXIF_FORMAT_UNDEFINED:
365  vs = entry->data[0];
366  break;
367  case EXIF_FORMAT_SHORT:
368  vs = exif_get_short(entry->data, entry->order);
369  break;
370  default:
371  vs = 0;
372  break;
373  }
374  /* find the value */
375  for (j = 0; items[i].elem[j].string &&
376  (items[i].elem[j].index < vs); j++);
377  if (items[i].elem[j].index != vs) {
378  snprintf (v, maxlen, _("Unknown value %hi"), vs);
379  break;
380  }
381  strncpy (v, items[i].elem[j].string, maxlen);
382  break;
383 
384  case MNOTE_NIKON_TAG_LENS:
385  CF (entry->format, EXIF_FORMAT_RATIONAL, v, maxlen);
386  CC (entry->components, 4, v, maxlen);
387  {
388  double c,d;
389  unsigned long a,b;
390  vr = exif_get_rational (entry->data, entry->order);
391  a = vr.numerator / vr.denominator;
392  vr = exif_get_rational (entry->data+8, entry->order);
393  b = vr.numerator / vr.denominator;
394  vr = exif_get_rational (entry->data+16, entry->order);
395  c = (double)vr.numerator / vr.denominator;
396  vr = exif_get_rational (entry->data+24, entry->order);
397  d = (double)vr.numerator / vr.denominator;
398  //printf("numerator %li, denominator %li\n", vr.numerator, vr.denominator);
399  snprintf (v, maxlen, "%ld-%ldmm 1:%3.1f - %3.1f",a,b,c,d);
400  }
401  break;

403  CF (entry->format, EXIF_FORMAT_RATIONAL, v, maxlen);

406  if (!vr.denominator) {
407  strncpy (v, _("Infinite"), maxlen);
408  } else {
409  r = (double)vr.numerator / vr.denominator;
410  snprintf (v, maxlen, "%2.2f", r);
411  }
412  break;

422  break;

425  break;

428  break;

445  break;
446  case 3:
447  strncat (v, _("bottom to top"), maxlen - strlen (v));
448  break;
449  case 4:
450  strncat (v, _("top to bottom"), maxlen - strlen (v));
451  break;
452  default:
453  snprintf (buf, sizeof (buf), _("%li"),
454  (long int) vl);
455  strncat (v, buf, maxlen - strlen (v));
456  }
457  break;
458  case MNOTE_OLYMPUS_TAG_UNKNOWN_1:
459  CF (entry->format, EXIF_FORMAT_SHORT, v, maxlen);
460  CC (entry->components, 1, v, maxlen);
461  strncpy (v, _("Unknown tag."), maxlen);
462  break;
463  case MNOTE_OLYMPUS_TAG_UNKNOWN_2:
464  CF (entry->format, EXIF_FORMAT_RATIONAL, v, maxlen);
465  CC (entry->components, 1, v, maxlen);
466  break;
467  case MNOTE_OLYMPUS_TAG_UNKNOWN_3:
468  CF (entry->format, EXIF_FORMAT_SSHORT, v, maxlen);
469  CC (entry->components, 1, v, maxlen);
470  break;
471  case MNOTE_OLYMPUS_TAG_VERSION:
472  CF (entry->format, EXIF_FORMAT_ASCII, v, maxlen);
473  CC2 (entry->components, 5, 8, v, maxlen);
474  strncpy (v, (char *)entry->data, MIN (maxlen, entry->size));
475  break;
476  case MNOTE_OLYMPUS_TAG_INFO:
477  CF (entry->format, EXIF_FORMAT_ASCII, v, maxlen);
478  CC2 (entry->components, 52, 60, v, maxlen);
479  strncpy (v, (char *)entry->data, MIN (maxlen, entry->size));
480  break;
481  case MNOTE_OLYMPUS_TAG_ID:
482  CF (entry->format, EXIF_FORMAT_UNDEFINED, v, maxlen);
483  CC (entry->components, 32, v, maxlen);
484  strncpy (v, (char *)entry->data, MIN (maxlen, entry->size));
485  break;
486  case MNOTE_OLYMPUS_TAG_UNKNOWN_4:
487  CF (entry->format, EXIF_FORMAT_LONG, v, maxlen);
488  CC (entry->components, 30, v, maxlen);
489  break;
490  case MNOTE_OLYMPUS_TAG_FOCUSDIST:
491  CF (entry->format, EXIF_FORMAT_RATIONAL, v, maxlen);
492  CC (entry->components, 1, v, maxlen);
493  vr = exif_get_rational (entry->data, entry->order);
494  if (vr.numerator == 0) {
495  strncpy (v, _("Unknown"), maxlen);
496  }
497  else {
498  unsigned long tmp = vr.numerator / vr.denominator;
499  /* printf("numerator %li, denominator %li\n", vr.numerator, vr.denominator); */
500  snprintf (v, maxlen, "%li mm", tmp);
501  }
502  break;
503  case MNOTE_OLYMPUS_TAG_WBALANCE:
504  CF (entry->format, EXIF_FORMAT_SHORT, v, maxlen);
505  CC (entry->components, 2, v, maxlen);
506  vs = exif_get_short (entry->data, entry->order);
507  switch (vs) {
508  case 1:
509  strncpy (v, _("Automatic"), maxlen);
510  break;
511  case 2:
512  {
513  ExifShort v2 = exif_get_short (entry->data + 2, entry->order);
514  unsigned long colorTemp = 0;
515  switch (v2) {
516  case 2:
517  colorTemp = 3000;
518  break;
519  case 3:
520  colorTemp = 3700;
521  break;
522  case 4:
523  colorTemp = 4000;
524  break;
525  case 5:
526  colorTemp = 4500;
527  break;
528  case 6:
529  colorTemp = 5500;
530  break;
531  case 7:
532  colorTemp = 6500;
533  break;
534  case 9:
535  colorTemp = 7500;
536  break;
