131 void
132 exif_content_add_entry (ExifContent *c, ExifEntry *entry)
133 {
134  ExifEntry **entries;
135  if (!c || !c->priv || !entry || entry->parent) return;
136 
137  /* One tag can only be added once to an IFD. */
138  if (exif_content_get_entry (c, entry->tag)) {
139  exif_log (c->priv->log, EXIF_LOG_CODE_DEBUG, "ExifContent",
140  "An attempt has been made to add "
141  "the tag '%s' twice to an IFD. This is against "
142  "specification.", exif_tag_get_name (entry->tag));
143  return;
144  }
145 
146  entries = exif_mem_realloc (c->priv->mem,
147  c->entries, sizeof (ExifEntry*) * (c->count + 1));
148  if (!entries) return;
149  entry->parent = c;
150  entries[c->count++] = entry;
151  c->entries = entries;
152  exif_entry_ref (entry);
153 }

155 void
156 exif_content_remove_entry (ExifContent *c, ExifEntry *e)
157 {
158  unsigned int i;
159 
160  if (!c || !c->priv || !e || (e->parent != c)) return;
161 
162  /* Search the entry */
163  for (i = 0; i < c->count; i++) if (c->entries[i] == e) break;
164  if (i == c->count) return;
165 
166  /* Remove the entry */
167  memmove (&c->entries[i], &c->entries[i + 1],
168  sizeof (ExifEntry*) * (c->count - i - 1));
169  c->count--;
170  e->parent = NULL;
171  exif_entry_unref (e);
172  c->entries = exif_mem_realloc (c->priv->mem, c->entries,
173  sizeof(ExifEntry*) * c->count);
174 }

234 void
235 exif_content_fix (ExifContent *c)
236 {
237  ExifIfd ifd = exif_content_get_ifd (c);
238  ExifDataType dt;
239  ExifTag t;
240  ExifEntry *e;
241 
242  if (!c) return;
243 
244  dt = exif_data_get_data_type (c->parent);
245 
246  /* First of all, fix all existing entries. */
247  exif_content_foreach_entry (c, fix_func, NULL);
248 
249  /*
250  * Then check for existing tags that are not allowed and for
251  * non-existing mandatory tags.
252  */
253  for (t = 0; t <= 0xffff; t++) {
254  switch (exif_tag_get_support_level_in_ifd (t, ifd, dt)) {
255  case EXIF_SUPPORT_LEVEL_MANDATORY:
256  if (exif_content_get_entry (c, t)) break;
257  exif_log (c->priv->log, EXIF_LOG_CODE_DEBUG, "exif-content",
258  "Tag '%s' is mandatory in IFD '%s' and has therefore been added.",
259  exif_tag_get_name_in_ifd (t, ifd), exif_ifd_get_name (ifd));
260  e = exif_entry_new ();
261  exif_content_add_entry (c, e);
262  exif_entry_initialize (e, t);
263  exif_entry_unref (e);
264  break;
265  case EXIF_SUPPORT_LEVEL_NOT_RECORDED:
266  e = exif_content_get_entry (c, t);
267  if (!e) break;
268  exif_log (c->priv->log, EXIF_LOG_CODE_DEBUG, "exif-content",
269  "Tag '%s' is not recoreded in IFD '%s' and has therefore been "
270  "removed.", exif_tag_get_name_in_ifd (t, ifd),
271  exif_ifd_get_name (ifd));
272  exif_content_remove_entry (c, e);
273  break;
274  case EXIF_SUPPORT_LEVEL_OPTIONAL:
275  default:
276  break;
277  }
278  }
