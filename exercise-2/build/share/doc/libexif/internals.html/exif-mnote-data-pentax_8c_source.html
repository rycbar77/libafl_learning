35 {

47  exif_mem_free (d->mem, n->entries);

71 static void
72 exif_mnote_data_pentax_load (ExifMnoteData *en,
73  const unsigned char *buf, unsigned int buf_size)
74 {
75  ExifMnoteDataPentax *n = (ExifMnoteDataPentax *) en;
76  unsigned int i, o, s;
77  ExifShort c;
78 
79  /* Number of entries */
80  if (buf_size < 2) return;
81  c = exif_get_short (buf + 6 + n->offset, n->order);
82  n->entries = exif_mem_alloc (en->mem, sizeof (MnotePentaxEntry) * c);
83  if (!n->entries) return;
84 
85  for (i = 0; i < c; i++) {
86  o = 6 + 2 + n->offset + 12 * i;
87  if (o + 8 > buf_size) return;
88 
89  n->count = i + 1;
90  n->entries[i].tag = exif_get_short (buf + o + 0, n->order);
91  n->entries[i].format = exif_get_short (buf + o + 2, n->order);
92  n->entries[i].components = exif_get_long (buf + o + 4, n->order);
93  n->entries[i].order = n->order;
94 
95  /*
96  * Size? If bigger than 4 bytes, the actual data is not
97  * in the entry but somewhere else (offset).
98  */
99  s = exif_format_get_size (n->entries[i].format) *
100  n->entries[i].components;
101  if (!s) return;
102  o += 8;
103  if (s > 4) o = exif_get_long (buf + o, n->order) + 6;
104  if (o + s > buf_size) return;
105 
106  /* Sanity check */
107  n->entries[i].data = exif_mem_alloc (en->mem, sizeof (char) * s);
108  if (!n->entries[i].data) return;
109  n->entries[i].size = s;
110  memcpy (n->entries[i].data, buf + o, s);
111  }
112 }
