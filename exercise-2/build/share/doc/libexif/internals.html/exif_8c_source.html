131 /*
132  * The name gets copied, so you can pass a static string;
133  * the data is not copied, so if it is a string,
134  * you must allocate it yourself.
135  */
136 static int
137 exif_append_data(exif_data_t **d_p,
138  char *name,
139  char rec_type,
140  int exif_format,
141  exif_rec_data_t *data)
142 {
143  exif_data_t *d = *d_p;
144 
145  if (rec_type == '\0')
146  return EXIT_FAILURE;
147 
148  if (d->n_alloc <= d->n_recs) {
149  d->n_alloc += EXIF_ALLOC_SIZE;
150  d = (*exif_realloc_fn)(d, sizeof(exif_data_t) +
151  sizeof(exif_record_t) * d->n_alloc);
152  *d_p = d;
153  }
154  d->recs[d->n_recs].rec_type = rec_type;
155  bcopy(data, &d->recs[d->n_recs].rec_data, sizeof(exif_rec_data_t));
156  d->recs[d->n_recs].rec_name = strdup(name);
157  d->n_recs++;
158  return EXIT_SUCCESS;
159 }

171 /*
172  Process a COM marker.
173  We want to print out the marker contents as legible text;
174  we must guard against random junk and varying newline representations.
175 */
176 static void
177 process_COM (ImageInfoType *ImageInfo, uchar *Data, int length)
178 {
179  int ch;
180  char *Comment;
181  int nch;
182  int a;
183  exif_rec_data_t rd;
184 
185  nch = 0;
186  Comment = (*exif_malloc_fn)(length+1);
187 
188  for (a=2;a<length;a++) {
189  ch = Data[a];
190 
191  if (ch == '\r' && Data[a+1] == '\n') continue; /* Remove cr followed by lf. */
192 
193  if (isprint(ch) || ch == '\n' || ch == '\t') {
194  Comment[nch++] = (char)ch;
195  } else {
196  Comment[nch++] = '?';
197  }
198  }
199 
200  Comment[nch] = '\0'; /* Null terminate */
201 
202  rd.s = Comment;
203  exif_append_data(&ImageInfo->d, "Comment", 's', EXIF_FMT_COMPUTED, &rd);
204 }

206 /* Process a SOFn marker. This is useful for the image dimensions. */
207 static void
208 process_SOFn (ImageInfoType *ImageInfo, uchar *Data, int marker)
209 {
210  int data_precision, num_components;
211  const char *process;
212  exif_rec_data_t rd;
213 
214  data_precision = Data[2];
215  rd.l = Get16m(Data+3);
216  exif_append_data(&ImageInfo->d,
217  "Height",
218  'l',
219  EXIF_FMT_COMPUTED,
220  &rd);
221  rd.l = Get16m(Data+5);
222  exif_append_data(&ImageInfo->d,
223  "Width",
224  'l',
225  EXIF_FMT_COMPUTED,
226  &rd);
227  num_components = Data[7];
228 
229  if (num_components == 3) {
230  rd.l = 1;
231  } else {
232  rd.l = 0;
233  }
234  exif_append_data(&ImageInfo->d, "IsColor", 'l', EXIF_FMT_COMPUTED, &rd);
235 
236  switch (marker) {
237  case M_SOF0: process = "Baseline"; break;
238  case M_SOF1: process = "Extended sequential"; break;
239  case M_SOF2: process = "Progressive"; break;
240  case M_SOF3: process = "Lossless"; break;
241  case M_SOF5: process = "Differential sequential"; break;
242  case M_SOF6: process = "Differential progressive"; break;
243  case M_SOF7: process = "Differential lossless"; break;
244  case M_SOF9: process = "Extended sequential, arithmetic coding"; break;
245  case M_SOF10: process = "Progressive, arithmetic coding"; break;
246  case M_SOF11: process = "Lossless, arithmetic coding"; break;
247  case M_SOF13: process = "Differential sequential, arithmetic coding"; break;
248  case M_SOF14: process = "Differential progressive, arithmetic coding"; break;
249  case M_SOF15: process = "Differential lossless, arithmetic coding"; break;
250  default: process = "Unknown"; break;
251  }
252 }

318 /* Convert a 16 bit unsigned value from file's native byte order */
319 static int
320 Get16u(void *Short, int MotorolaOrder)
321 {
322  if (MotorolaOrder) {
323  return (((uchar *)Short)[0] << 8) | ((uchar *)Short)[1];
324  } else {
325  return (((uchar *)Short)[1] << 8) | ((uchar *)Short)[0];
326  }
327 }
328 
329 /* Convert a 32 bit signed value from file's native byte order */
330 static int
331 Get32s(void *Long, int MotorolaOrder)
332 {
333  if (MotorolaOrder) {
334  return ((( char *)Long)[0] << 24) | (((uchar *)Long)[1] << 16)
335  | (((uchar *)Long)[2] << 8 ) | (((uchar *)Long)[3] << 0 );
336  } else {
337  return ((( char *)Long)[3] << 24) | (((uchar *)Long)[2] << 16)
338  | (((uchar *)Long)[1] << 8 ) | (((uchar *)Long)[0] << 0 );
339  }
340 }
341 
342 /* Convert a 32 bit unsigned value from file's native byte order */
343 static unsigned
344 Get32u(void *Long, int MotorolaOrder)
345 {
346  return (unsigned)Get32s(Long, MotorolaOrder) & 0xffffffff;
347 }

350 /* Evaluate number, be it int, rational, or float from directory. */
351 static double
352 ConvertAnyFormat(void *ValuePtr, int Format, int MotorolaOrder)
353 {
354  double Value;
355  Value = 0;
356 
357  switch(Format) {
358  case EXIF_FMT_SBYTE: Value = *(signed char *)ValuePtr; break;
359  case EXIF_FMT_BYTE: Value = *(uchar *)ValuePtr; break;
360 
361  case EXIF_FMT_USHORT: Value = Get16u(ValuePtr,MotorolaOrder); break;
362  case EXIF_FMT_ULONG: Value = Get32u(ValuePtr,MotorolaOrder); break;
363 
364  case EXIF_FMT_URATIONAL:
365  case EXIF_FMT_SRATIONAL:
366  {
367  int Num,Den;
368  Num = Get32s(ValuePtr,MotorolaOrder);
369  Den = Get32s(4+(char *)ValuePtr,MotorolaOrder);
370  if (Den == 0) {
371  Value = 0;
372  } else {
373  Value = (double)Num/Den;
374  }
375  break;
376  }
377 
378  case EXIF_FMT_SSHORT: Value = (signed short)Get16u(ValuePtr,MotorolaOrder); break;
379  case EXIF_FMT_SLONG: Value = Get32s(ValuePtr,MotorolaOrder); break;
380 
381  /* Not sure if this is correct (never seen float used in Exif format) */
382  case EXIF_FMT_SINGLE: Value = (double)*(float *)ValuePtr; break;
383  case EXIF_FMT_DOUBLE: Value = *(double *)ValuePtr; break;
384  }
385  return Value;
386 }

388 /* Evaluate number, be it int, rational, or float from directory. */
389 static char
390 ConvertAnyFormat2(void *ValuePtr, int ByteCount, int Format, int MotorolaOrder, exif_rec_data_t *data_p)
391 {
392  char *str, *p;
393  char r_type;
394  unsigned char c;
395  static char hexdigits[] = "0123456789ABCDEF";
396 
397  switch(Format) {
398  case EXIF_FMT_STRING:
399  data_p->s = exif_strndup(ValuePtr, ByteCount);
400  r_type = 's';
401  break;
402 
403  case EXIF_FMT_SBYTE:
404  data_p->l = (long)*(signed char *)ValuePtr;
405  r_type = 'l';
406  break;
407 
408  case EXIF_FMT_BYTE:
409  data_p->l = (long)*(uchar *)ValuePtr;
410  r_type = 'l';
411  break;
412 
413  case EXIF_FMT_USHORT:
414  data_p->l = (long)Get16u(ValuePtr,MotorolaOrder);
415  r_type = 'l';
416  break;
417  case EXIF_FMT_ULONG:
418  data_p->l = (long)Get32u(ValuePtr,MotorolaOrder);
419  r_type = 'l';
420  break;
421 
422  case EXIF_FMT_URATIONAL:
423  case EXIF_FMT_SRATIONAL:
424  {
425  int Num,Den;
426  data_p->r.num = Get32s(ValuePtr,MotorolaOrder);
427  data_p->r.denom = Get32s(4+(char *)ValuePtr,MotorolaOrder);
428  r_type = 'r';
429  break;
430  }
431 
432  case EXIF_FMT_SSHORT:
433  data_p->l = (signed short)Get16u(ValuePtr,MotorolaOrder);
434  r_type = 'l';
435  break;
436  case EXIF_FMT_SLONG:
437  data_p->l = (long)Get32s(ValuePtr,MotorolaOrder);
438  r_type = 'l';
439  break;
440 
441  /* Not sure if this is correct (never seen float used in Exif format) */
442  case EXIF_FMT_SINGLE:
443  data_p->f = *(float *)ValuePtr;
444  r_type = 'f';
445  break;
446 
447  case EXIF_FMT_DOUBLE:
448  data_p->g = *(double *)ValuePtr;
449  r_type = 'f';
450  break;
451 
452  default:
453  /* unknown type */
454  p = str = (*exif_malloc_fn)(ByteCount*2 + 1);
455  while (ByteCount--) {
456  c = *(unsigned char *)ValuePtr++;
457  *p++ = hexdigits[c / 16];
458  *p++ = hexdigits[c % 16];
459  }
460  *p++ = '\0';
461  data_p->s = str;
462  r_type = 's';
463  break;
464  }
465  return r_type;
466 }

525 static void
526 ProcessUserComment(ImageInfoType *ImageInfo,
527  void *_ValuePtr,
528  int ByteCount,
529  int Format,
530  struct ExifTag *tag_p)
531 {
532  char *ValuePtr = (char *)_ValuePtr;
533  exif_rec_data_t rd;
534  int a;
535 
536  /* Olympus has this padded with trailing spaces. Remove these first. */
537  for (a=ByteCount;;) {
538  a--;
539  if ((ValuePtr)[a] == ' ') {
540  (ValuePtr)[a] = '\0';
541  } else {
542  break;
543  }
544  if (a == 0) break;
545  }
546 
547  /* Copy the comment */
548  if (memcmp(ValuePtr, "ASCII",5) == 0) {
549  for (a=5;a<10;a++) {
550  int c;
551  c = (ValuePtr)[a];
552  if (c != '\0' && c != ' ') {
553  rd.s = exif_strndup(a+ValuePtr, ByteCount - a);
554  exif_append_data(&ImageInfo->d,
555  "UserComment",
556  's',
557  Format,
558  &rd);
559  break;
560  }
561  }
562 
563  } else {
564  rd.s = exif_strndup(ValuePtr, ByteCount);
565  exif_append_data(&ImageInfo->d,
566  "UserComment",
567  's',
568  Format,
569  &rd);
570  }

817 /* Process one of the nested EXIF directories. */
818 static int
819 ProcessExifDir(ImageInfoType *ImageInfo, char *DirStart, char *OffsetBase, unsigned ExifLength, char *LastExifRefd)
820 {
821  int de;
822  int NumDirEntries;
823  exif_rec_data_t rd;
824  char rec_type;
825  char label[32];
826 
827  NumDirEntries = Get16u(DirStart, ImageInfo->MotorolaOrder);
828 
829  if ((DirStart+2+NumDirEntries*12) > (OffsetBase+ExifLength)) {
830  exif_error("Illegally sized directory");
831  return FALSE;
832  }
833 
834 
835  for (de=0;de<NumDirEntries;de++) {
836  int Tag, Format, Components;
837  char *ValuePtr;
838  int ByteCount;
839  char *DirEntry;
840  struct ExifTag *tag_p;
841 
842  DirEntry = DirStart+2+12*de;
843 
844  Tag = Get16u(DirEntry, ImageInfo->MotorolaOrder);
845  Format = Get16u(DirEntry+2, ImageInfo->MotorolaOrder);
846  Components = Get32u(DirEntry+4, ImageInfo->MotorolaOrder);
847 
848  if ((Format-1) >= NUM_FORMATS) {
849  /* (-1) catches illegal zero case as unsigned underflows to positive large. */
850  exif_error("Illegal format code in EXIF dir");
851  return FALSE;
852  }
853 
854  ByteCount = Components * ExifBytesPerFormat[Format];
855 
856  if (ByteCount > 4) {
857  unsigned OffsetVal;
858  OffsetVal = Get32u(DirEntry+8, ImageInfo->MotorolaOrder);
859  /* If its bigger than 4 bytes, the dir entry contains an offset. */
860  if (OffsetVal+ByteCount > ExifLength) {
861  /* Bogus pointer offset and / or bytecount value */
862  /* printf("Offset %d bytes %d ExifLen %d\n",OffsetVal, ByteCount, ExifLength); */
863 
864  exif_error("Illegal pointer offset value in EXIF");
865  return FALSE;
866  }
867  ValuePtr = OffsetBase+OffsetVal;
868  } else {
869  /* 4 bytes or less and value is in the dir entry itself */
870  ValuePtr = DirEntry+8;
871  }
872 
873  if (LastExifRefd < ValuePtr+ByteCount) {
874  /*
875  Keep track of last byte in the exif header that was actually referenced.
876  That way, we know where the discardable thumbnail data begins.
877  */
878  LastExifRefd = ValuePtr+ByteCount;
879  }
880 
881  if (Tag == TAG_EXIF_OFFSET || Tag == TAG_INTEROP_OFFSET) {
882  char *SubdirStart;
883  SubdirStart = OffsetBase + Get32u(ValuePtr, ImageInfo->MotorolaOrder);
884  if (SubdirStart < OffsetBase || SubdirStart > OffsetBase+ExifLength) {
885  exif_error("Illegal subdirectory link");
886  return FALSE;
887  }
888  ProcessExifDir(ImageInfo, SubdirStart, OffsetBase, ExifLength, LastExifRefd);
889  continue;
890  }
891 
892  /* Search through tag table */
893  for (tag_p = &TagTable[0]; tag_p->Desc != NULL; tag_p++) {
894  if (tag_p->Tag == Tag) {
895  if (tag_p->Func != NULL) {
896  (*tag_p->Func)(ImageInfo, ValuePtr, ByteCount, Format, tag_p, OffsetBase);
897  } else {
898  rec_type = ConvertAnyFormat2(ValuePtr, ByteCount, Format,
899  ImageInfo->MotorolaOrder,
900  &rd);
901  exif_append_data(&ImageInfo->d,
902  tag_p->Desc,
903  rec_type,
904  Format,
905  &rd);
906  }
907  break;
908  }
909  }
910  if (tag_p->Desc == NULL) {
911  rec_type = ConvertAnyFormat2(ValuePtr, ByteCount, Format,
912  ImageInfo->MotorolaOrder,
913  &rd);
914  snprintf(label, 32, "0x%04x", (unsigned int) Tag);
915  exif_append_data(&ImageInfo->d,
916  label,
917  rec_type,
918  Format,
919  &rd);
920  }
921  }
922  return TRUE;
923 }

925 /*
926  Process an EXIF marker
927  Describes all the drivel that most digital cameras include...
928 */
929 static int
930 process_EXIF (ImageInfoType *ImageInfo, char *CharBuf, unsigned int length, char *LastExifRefd)
931 {
932  int cc;
933  LastExifRefd = CharBuf;
934 
935  { /* Check the EXIF header component */
936  static const uchar ExifHeader[] = {0x45, 0x78, 0x69, 0x66, 0x00, 0x00};
937  if (memcmp(CharBuf+2, ExifHeader,6)) {
938  exif_error("Incorrect Exif header");
939  return FALSE;
940  }
941  }
942 
943  if (memcmp(CharBuf+8,"II",2) == 0) {
944  ImageInfo->MotorolaOrder = 0;
945  } else {
946  if (memcmp(CharBuf+8,"MM",2) == 0) {
947  ImageInfo->MotorolaOrder = 1;
948  } else {
949  exif_error("Invalid Exif alignment marker.");
950  return FALSE;
951  }
952  }
953 
954  /* Check the next two values for correctness. */
955  if (Get16u(CharBuf+10,ImageInfo->MotorolaOrder) != 0x2a
956  || Get32u(CharBuf+12,ImageInfo->MotorolaOrder) != 0x08) {
957  exif_error("Invalid Exif start (1, NULL)");
958  return FALSE;
959  }
960 
961  /* First directory starts 16 bytes in. Offsets start at 8 bytes in. */
962  cc = ProcessExifDir(ImageInfo, CharBuf+16, CharBuf+8, length-6, LastExifRefd);
963  if (cc != TRUE) {
964  return cc;
965  }
966  return TRUE;
967 }

969 /* Parse the marker stream until SOS or EOI is seen; */
970 static int
971 scan_JPEG_header (ImageInfoType *ImageInfo, FILE *infile, Section_t *Sections, int *SectionsRead, int ReadAll, char *LastExifRefd)
972 {
973  int a;
974  int HaveCom = FALSE;
975 
976  a = fgetc(infile);
977  if (a != 0xff || fgetc(infile) != M_SOI) {
978  return FALSE;
979  }
980 
981  for(*SectionsRead=0;*SectionsRead < 19;) {
982  int itemlen;
983  int marker = 0;
984  int ll,lh;
985  size_t got;
986  uchar *Data;
987 
988  for (a=0;a<7;a++) {
989  marker = fgetc(infile);
990  if (marker != 0xff) break;
991  }
992  if (marker == 0xff) {
993  /* 0xff is legal padding, but if we get that many, something's wrong. */
994  exif_error("too many padding bytes!");
995  return FALSE;
996  }
997 
998  Sections[*SectionsRead].Type = marker;
999 
1000  /* Read the length of the section. */
1001  lh = fgetc(infile);
1002  ll = fgetc(infile);
1003 
1004  itemlen = (lh << 8) | ll;
1005 
1006  if (itemlen < 2) {
1007  exif_error("invalid marker");
1008  return FALSE;
1009  }
1010 
1011  Sections[*SectionsRead].Size = (unsigned int) itemlen;
1012 
1013  Data = (uchar *)(*exif_malloc_fn)(itemlen+1); /* Add 1 to allow sticking a 0 at the end. */
1014  Sections[*SectionsRead].Data = Data;
1015 
1016  /* Store first two pre-read bytes. */
1017  Data[0] = (uchar)lh;
1018  Data[1] = (uchar)ll;
1019 
1020  got = fread(Data+2, 1, (size_t) itemlen-2, infile); /* Read the whole section. */
1021  if ((int) got != itemlen-2) {
1022  exif_error("reading from file");
1023  return FALSE;
1024  }
1025  *SectionsRead += 1;
1026 
1027  switch(marker) {
1028  case M_SOS: /* stop before hitting compressed data */
1029  /* If reading entire image is requested, read the rest of the data. */
1030  if (ReadAll) {
1031  signed long cp, ep;
1032  unsigned int size;
1033  /* Determine how much file is left. */
1034  cp = ftell(infile);
1035  if (cp == -1) {
1036  exif_error ("ftell failed");
1037  return FALSE;
1038  }
1039  if (fseek(infile, 0, SEEK_END) == -1) {
1040  exif_error ("fseek failed");
1041  return FALSE;
1042  }
1043  ep = ftell(infile);
1044  if (ep == -1) {
1045  exif_error ("ftell failed");
1046  return FALSE;
1047  }
1048  if (fseek(infile, cp, SEEK_SET) == -1) {
1049  exif_error ("fseek failed");
1050  return FALSE;
1051  }
1052 
1053  /* a very basic range check */
1054  if (ep <= cp ) {
1055  exif_error ("range check failed");
1056  return FALSE;
1057  }
1058 
1059  size = (unsigned int) (ep-cp);
1060  Data = (uchar *)(*exif_malloc_fn)((int)size);
1061  if (Data == NULL) {
1062  exif_error("could not allocate data for entire image");
1063  return FALSE;
1064  }
1065 
1066  got = fread(Data, 1, size, infile);
1067  if (got != size) {
1068  exif_error("could not read the rest of the image");
1069  return FALSE;
1070  }
1071 
1072  Sections[*SectionsRead].Data = Data;
1073  Sections[*SectionsRead].Size = size;
1074  Sections[*SectionsRead].Type = PSEUDO_IMAGE_MARKER;
1075  (*SectionsRead)++;
1076  /*
1077  *HaveAll = 1;
1078  */
1079  }
1080  return TRUE;
1081 
1082  case M_EOI: /* in case it's a tables-only JPEG stream */
1083  exif_error("No image in jpeg!");
1084  return FALSE;
1085 
1086  case M_COM: /* Comment section */
1087  if (HaveCom) {
1088  (*SectionsRead) -= 1;
1089  (*exif_free_fn)(Sections[*SectionsRead].Data);
1090  } else {
1091  process_COM(ImageInfo, Data, itemlen);
1092  HaveCom = TRUE;
1093  }
1094  break;
1095 
1096  case M_EXIF:
1097  if (*SectionsRead <= 2) {
1098  /* Seen files from some 'U-lead' software with Vivitar scanner
1099  that uses marker 31 later in the file (no clue what for!) */
1100  process_EXIF(ImageInfo, (char *)Data, (unsigned int) itemlen, LastExifRefd);
1101  }
1102  break;
1103 
1104  case M_SOF0:
1105  case M_SOF1:
1106  case M_SOF2:
1107  case M_SOF3:
1108  case M_SOF5:
1109  case M_SOF6:
1110  case M_SOF7:
1111  case M_SOF9:
1112  case M_SOF10:
1113  case M_SOF11:
1114  case M_SOF13:
1115  case M_SOF14:
1116  case M_SOF15:
1117  process_SOFn(ImageInfo, Data, marker);
1118  break;
1119  default:
1120  /* skip any other marker silently. */
1121  break;
1122  }
1123  }
1124  return TRUE;
1125 }
