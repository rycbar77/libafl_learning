155 static int
156 exif_data_load_data_entry (ExifData *data, ExifEntry *entry,
157  const unsigned char *d,
158  unsigned int size, unsigned int offset)
159 {
160  unsigned int s, doff;
161 
162  entry->tag = exif_get_short (d + offset + 0, data->priv->order);
163  entry->format = exif_get_short (d + offset + 2, data->priv->order);
164  entry->components = exif_get_long (d + offset + 4, data->priv->order);
165 
166  exif_log (data->priv->log, EXIF_LOG_CODE_DEBUG, "ExifData",
167  "Loading entry 0x%x ('%s')...", entry->tag,
168  exif_tag_get_name (entry->tag));
169 
170  /*
171  * Size? If bigger than 4 bytes, the actual data is not
172  * in the entry but somewhere else (offset).
173  */
174  s = exif_format_get_size (entry->format) * entry->components;
175  if (!s)
176  return 0;
177  if (s > 4)
178  doff = exif_get_long (d + offset + 8, data->priv->order);
179  else
180  doff = offset + 8;
181 
182  /* Sanity checks */
183  if ((doff + s < doff) || (doff + s < s))
184  return 0;
185  if (size < doff + s)
186  return 0;
187 
188  entry->data = exif_data_alloc (data, s);
189  if (entry->data) {
190  entry->size = s;
191  memcpy (entry->data, d + doff, s);
192  }
193 
194  /* If this is the MakerNote, remember the offset */
195  if (entry->tag == EXIF_TAG_MAKER_NOTE) {
196  if (entry->size > 6) exif_log (data->priv->log,
197  EXIF_LOG_CODE_DEBUG, "ExifData",
198  "MakerNote found (%02x %02x %02x %02x "
199  "%02x %02x %02x...).",
200  entry->data[0], entry->data[1], entry->data[2],
201  entry->data[3], entry->data[4], entry->data[5],
202  entry->data[6]);
203  data->priv->offset_mnote = doff;
204  }
205  return 1;
206 }

208 static void
209 exif_data_save_data_entry (ExifData *data, ExifEntry *e,
210  unsigned char **d, unsigned int *ds,
211  unsigned int offset)
212 {
213  unsigned int doff, s;
214 
215  if (!data || !data->priv)
216  return;
217 
218  /*
219  * Each entry is 12 bytes long. The memory for the entry has
220  * already been allocated.
221  */
222  exif_set_short (*d + 6 + offset + 0,
223  data->priv->order, (ExifShort) e->tag);
224  exif_set_short (*d + 6 + offset + 2,
225  data->priv->order, (ExifShort) e->format);
226 
227  if (!(data->priv->options & EXIF_DATA_OPTION_DONT_CHANGE_MAKER_NOTE)) {
228  /* If this is the maker note tag, update it. */
229  if ((e->tag == EXIF_TAG_MAKER_NOTE) && data->priv->md) {
230  exif_mem_free (data->priv->mem, e->data);
231  e->data = NULL;
232  e->size = 0;
233  exif_mnote_data_set_offset (data->priv->md, *ds - 6);
234  exif_mnote_data_save (data->priv->md, &e->data, &e->size);
235  e->components = e->size;
236  }
237  }
238 
239  exif_set_long (*d + 6 + offset + 4,
240  data->priv->order, e->components);
241 
242  /*
243  * Size? If bigger than 4 bytes, the actual data is not in
244  * the entry but somewhere else.
245  */
246  s = exif_format_get_size (e->format) * e->components;
247  if (s > 4) {
248  doff = *ds - 6;
249  *ds += s;
250 
251  /*
252  * According to the TIFF specification,
253  * the offset must be an even number. If we need to introduce
254  * a padding byte, we set it to 0.
255  */
256  if (s & 1)
257  (*ds)++;
258  *d = exif_mem_realloc (data->priv->mem, *d, *ds);
259  if (!*d) {
260  EXIF_LOG_NO_MEMORY (data->priv->log, "ExifData", *ds);
261  return;
262  }
263  exif_set_long (*d + 6 + offset + 8, data->priv->order, doff);
264  if (s & 1)
265  *(*d + *ds - 1) = '\0';
266 
267  } else
268  doff = offset + 8;
269 
270  /* Write the data. Fill unneeded bytes with 0. */
271  memcpy (*d + 6 + doff, e->data, s);
272  if (s < 4)
273  memset (*d + 6 + doff + s, 0, (4 - s));
274 }

313 static void
314 exif_data_load_data_content (ExifData *data, ExifIfd ifd,
315  const unsigned char *d,
316  unsigned int ds, unsigned int offset, unsigned int recursion_depth)
317 {
318  ExifLong o, thumbnail_offset = 0, thumbnail_length = 0;
319  ExifShort n;
320  ExifEntry *entry;
321  unsigned int i;
322  ExifTag tag;
323 
324  if (!data || !data->priv)
325  return;
326  if ((ifd < 0) || (ifd >= EXIF_IFD_COUNT))
327  return;
328 
329  if (recursion_depth > 150) {
330  exif_log (data->priv->log, EXIF_LOG_CODE_CORRUPT_DATA, "ExifData",
331  "Deep recursion detected!");
332  return;
333  }
334 
335  /* Read the number of entries */
336  if (offset >= ds - 1)
337  return;
338  n = exif_get_short (d + offset, data->priv->order);
339  exif_log (data->priv->log, EXIF_LOG_CODE_DEBUG, "ExifData",
340  "Loading %i entries...", n);
341  offset += 2;
342 
343  /* Check if we have enough data. */
344  if (offset + 12 * n > ds)
345  n = (ds - offset) / 12;
346 
347  for (i = 0; i < n; i++) {
348 
349  tag = exif_get_short (d + offset + 12 * i, data->priv->order);
350  switch (tag) {
351  case EXIF_TAG_EXIF_IFD_POINTER:
352  case EXIF_TAG_GPS_INFO_IFD_POINTER:
353  case EXIF_TAG_INTEROPERABILITY_IFD_POINTER:
354  case EXIF_TAG_JPEG_INTERCHANGE_FORMAT_LENGTH:
355  case EXIF_TAG_JPEG_INTERCHANGE_FORMAT:
356  o = exif_get_long (d + offset + 12 * i + 8,
357  data->priv->order);
358  switch (tag) {
359  case EXIF_TAG_EXIF_IFD_POINTER:
360  CHECK_REC (EXIF_IFD_EXIF);
361  exif_data_load_data_content (data, EXIF_IFD_EXIF, d, ds, o, recursion_depth + 1);
362  break;
363  case EXIF_TAG_GPS_INFO_IFD_POINTER:
364  CHECK_REC (EXIF_IFD_GPS);
365  exif_data_load_data_content (data, EXIF_IFD_GPS, d, ds, o, recursion_depth + 1);
366  break;
367  case EXIF_TAG_INTEROPERABILITY_IFD_POINTER:
368  CHECK_REC (EXIF_IFD_INTEROPERABILITY);
369  exif_data_load_data_content (data, EXIF_IFD_INTEROPERABILITY, d, ds, o, recursion_depth + 1);
370  break;
371  case EXIF_TAG_JPEG_INTERCHANGE_FORMAT:
372  thumbnail_offset = o;
373  if (thumbnail_offset && thumbnail_length)
374  exif_data_load_data_thumbnail (data, d,
375  ds, thumbnail_offset,
376  thumbnail_length);
377  break;
378  case EXIF_TAG_JPEG_INTERCHANGE_FORMAT_LENGTH:
379  thumbnail_length = o;
380  if (thumbnail_offset && thumbnail_length)
381  exif_data_load_data_thumbnail (data, d,
382  ds, thumbnail_offset,
383  thumbnail_length);
384  break;
385  default:
386  return;
387  }
388  break;
389  default:
390 
391  /*
392  * If we don't know the tag, don't fail. It could be that new
393  * versions of the standard have defined additional tags. Note that
394  * 0 is a valid tag in the GPS IFD.
395  */
396  if (!exif_tag_get_name_in_ifd (tag, ifd)) {
397 
398  /*
399  * Special case: Tag and format 0. That's against specification.
400  * At least up to 2.2. But Photoshop writes it anyways.
401  */
402  if (!memcmp (d + offset + 12 * i, "\0\0\0\0", 4)) {
403  exif_log (data->priv->log, EXIF_LOG_CODE_DEBUG, "ExifData",
404  "Skipping empty entry at position %i in '%s'.", i,
405  exif_ifd_get_name (ifd));
406  break;
407  }
408  exif_log (data->priv->log, EXIF_LOG_CODE_DEBUG, "ExifData",
409  "Unknown tag 0x%04x (entry %i in '%s'). Please report this tag "
410  "to <libexif-devel@lists.sourceforge.net>.", tag, i,
411  exif_ifd_get_name (ifd));
412  if (data->priv->options & EXIF_DATA_OPTION_IGNORE_UNKNOWN_TAGS)
413  break;
414  }
415  entry = exif_entry_new_mem (data->priv->mem);
416  if (exif_data_load_data_entry (data, entry, d, ds,
417  offset + 12 * i))
418  exif_content_add_entry (data->ifd[ifd], entry);
419  exif_entry_unref (entry);
420  break;
421  }
422  }
423 }

448 static void
449 exif_data_save_data_content (ExifData *data, ExifContent *ifd,
450  unsigned char **d, unsigned int *ds,
451  unsigned int offset)
452 {
453  unsigned int j, n_ptr = 0, n_thumb = 0;
454  ExifIfd i;
455 
456  if (!data || !data->priv || !ifd || !d || !ds)
457  return;
458 
459  for (i = 0; i < EXIF_IFD_COUNT; i++)
460  if (ifd == data->ifd[i])
461  break;
462  if (i == EXIF_IFD_COUNT)
463  return;
464 
465  /*
466  * Check if we need some extra entries for pointers or the thumbnail.
467  */
468  switch (i) {
469  case EXIF_IFD_0:
470 
471  /*
472  * The pointer to IFD_EXIF is in IFD_0. The pointer to
473  * IFD_INTEROPERABILITY is in IFD_EXIF.
474  */
475  if (data->ifd[EXIF_IFD_EXIF]->count ||
476  data->ifd[EXIF_IFD_INTEROPERABILITY]->count)
477  n_ptr++;
478 
479  /* The pointer to IFD_GPS is in IFD_0. */
480  if (data->ifd[EXIF_IFD_GPS]->count)
481  n_ptr++;
482 
483  break;
484  case EXIF_IFD_1:
485  if (data->size)
486  n_thumb = 2;
487  break;
488  case EXIF_IFD_EXIF:
489  if (data->ifd[EXIF_IFD_INTEROPERABILITY]->count)
490  n_ptr++;
491  default:
492  break;
493  }
494 
495  /*
496  * Allocate enough memory for all entries
497  * and the number of entries.
498  */
499  *ds += (2 + (ifd->count + n_ptr + n_thumb) * 12 + 4);
500  *d = exif_mem_realloc (data->priv->mem, *d, *ds);
501  if (!*d) {
502  EXIF_LOG_NO_MEMORY (data->priv->log, "ExifData", *ds);
503  return;
504  }
505 
506  /* Save the number of entries */
507  exif_set_short (*d + 6 + offset, data->priv->order,
508  (ExifShort) (ifd->count + n_ptr + n_thumb));
509  offset += 2;
510 
511  /* Save each entry */
512  exif_log (data->priv->log, EXIF_LOG_CODE_DEBUG, "ExifData",
513  "Saving %i entries (IFD '%s', offset: %i)...",
514  ifd->count, exif_ifd_get_name (i), offset);
515  for (j = 0; j < ifd->count; j++)
516  exif_data_save_data_entry (data, ifd->entries[j], d, ds, offset + 12 * j);
517 
518  offset += 12 * ifd->count;
519 
520  /* Now save special entries. */
521  switch (i) {
522  case EXIF_IFD_0:
523 
524  /*
525  * The pointer to IFD_EXIF is in IFD_0.
526  * However, the pointer to IFD_INTEROPERABILITY is in IFD_EXIF,
527  * therefore, if IFD_INTEROPERABILITY is not empty, we need
528  * IFD_EXIF even if latter is empty.
529  */
530  if (data->ifd[EXIF_IFD_EXIF]->count ||
531  data->ifd[EXIF_IFD_INTEROPERABILITY]->count) {
532  exif_set_short (*d + 6 + offset + 0, data->priv->order,
533  EXIF_TAG_EXIF_IFD_POINTER);
534  exif_set_short (*d + 6 + offset + 2, data->priv->order,
535  EXIF_FORMAT_LONG);
536  exif_set_long (*d + 6 + offset + 4, data->priv->order,
537  1);
538  exif_set_long (*d + 6 + offset + 8, data->priv->order,
539  *ds - 6);
540  exif_data_save_data_content (data,
541  data->ifd[EXIF_IFD_EXIF], d, ds, *ds - 6);
542  offset += 12;
543  }
544 
545  /* The pointer to IFD_GPS is in IFD_0, too. */
546  if (data->ifd[EXIF_IFD_GPS]->count) {
547  exif_set_short (*d + 6 + offset + 0, data->priv->order,
548  EXIF_TAG_GPS_INFO_IFD_POINTER);
549  exif_set_short (*d + 6 + offset + 2, data->priv->order,
550  EXIF_FORMAT_LONG);
551  exif_set_long (*d + 6 + offset + 4, data->priv->order,
552  1);
553  exif_set_long (*d + 6 + offset + 8, data->priv->order,
554  *ds - 6);
555  exif_data_save_data_content (data,
556  data->ifd[EXIF_IFD_GPS], d, ds, *ds - 6);
557  offset += 12;
558  }
559 
560  break;
561  case EXIF_IFD_EXIF:
562 
563  /*
564  * The pointer to IFD_INTEROPERABILITY is in IFD_EXIF.
565  * See note above.
566  */
567  if (data->ifd[EXIF_IFD_INTEROPERABILITY]->count) {
568  exif_set_short (*d + 6 + offset + 0, data->priv->order,
569  EXIF_TAG_INTEROPERABILITY_IFD_POINTER);
570  exif_set_short (*d + 6 + offset + 2, data->priv->order,
571  EXIF_FORMAT_LONG);
572  exif_set_long (*d + 6 + offset + 4, data->priv->order,
573  1);
574  exif_set_long (*d + 6 + offset + 8, data->priv->order,
575  *ds - 6);
576  exif_data_save_data_content (data,
577  data->ifd[EXIF_IFD_INTEROPERABILITY], d, ds,
578  *ds - 6);
579  offset += 12;
580  }
581 
582  break;
583  case EXIF_IFD_1:
584 
585  /*
586  * Information about the thumbnail (if any) is saved in
587  * IFD_1.
588  */
589  if (data->size) {
590 
591  /* EXIF_TAG_JPEG_INTERCHANGE_FORMAT */
592  exif_set_short (*d + 6 + offset + 0, data->priv->order,
593  EXIF_TAG_JPEG_INTERCHANGE_FORMAT);
594  exif_set_short (*d + 6 + offset + 2, data->priv->order,
595  EXIF_FORMAT_LONG);
596  exif_set_long (*d + 6 + offset + 4, data->priv->order,
597  1);
598  exif_set_long (*d + 6 + offset + 8, data->priv->order,
599  *ds - 6);
600  *ds += data->size;
601  *d = exif_mem_realloc (data->priv->mem, *d, *ds);
602  if (!*d) {
603  EXIF_LOG_NO_MEMORY (data->priv->log, "ExifData",
604  *ds);
605  return;
606  }
607  memcpy (*d + *ds - data->size, data->data, data->size);
608  offset += 12;
609 
610  /* EXIF_TAG_JPEG_INTERCHANGE_FORMAT_LENGTH */
611  exif_set_short (*d + 6 + offset + 0, data->priv->order,
612  EXIF_TAG_JPEG_INTERCHANGE_FORMAT_LENGTH);
613  exif_set_short (*d + 6 + offset + 2, data->priv->order,
614  EXIF_FORMAT_LONG);
615  exif_set_long (*d + 6 + offset + 4, data->priv->order,
616  1);
617  exif_set_long (*d + 6 + offset + 8, data->priv->order,
618  data->size);
619  offset += 12;
620  }
621 
622  break;
623  default:
624  break;
625  }
626 
627  /* Sort the directory according to TIFF specification */
628  qsort (*d + 6 + offset - (ifd->count + n_ptr + n_thumb) * 12,
629  (ifd->count + n_ptr + n_thumb), 12,
630  (data->priv->order == EXIF_BYTE_ORDER_INTEL) ? cmp_func_intel : cmp_func_motorola);
631 
632  /* Correctly terminate the directory */

697 void
698 exif_data_load_data (ExifData *data, const unsigned char *d_orig,
699  unsigned int ds_orig)
700 {
701  unsigned int l;
702  ExifLong offset;
703  ExifShort n;
704  const unsigned char *d = d_orig;
705  unsigned int ds = ds_orig, len;
706 
707  if (!data || !data->priv || !d || !ds)
708  return;
709 
710  exif_log (data->priv->log, EXIF_LOG_CODE_DEBUG, "ExifData",
711  "Parsing %i byte(s) EXIF data...\n", ds);
712 
713  /*
714  * It can be that the data starts with the EXIF header. If it does
715  * not, search the EXIF marker.
716  */
717  if (ds < 6) {
718  LOG_TOO_SMALL;
719  return;
720  }
721  if (!memcmp (d, ExifHeader, 6)) {
722  exif_log (data->priv->log, EXIF_LOG_CODE_DEBUG, "ExifData",
723  "Found EXIF header.");
724  } else {
725  while (1) {
726  while ((d[0] == 0xff) && ds) {
727  d++;
728  ds--;
729  }
730 
731  /* JPEG_MARKER_SOI */
732  if (d[0] == JPEG_MARKER_SOI) {
733  d++;
734  ds--;
735  continue;
736  }
737 
738  /* JPEG_MARKER_APP0 */
739  if (d[0] == JPEG_MARKER_APP0) {
740  d++;
741  ds--;
742  l = (d[0] << 8) | d[1];
743  if (l > ds)
744  return;
745  d += l;
746  ds -= l;
747  continue;
748  }
749 
750  /* JPEG_MARKER_APP1 */
751  if (d[0] == JPEG_MARKER_APP1)
752  break;
753 
754  /* Unknown marker or data. Give up. */
755  exif_log (data->priv->log, EXIF_LOG_CODE_CORRUPT_DATA,
756  "ExifData", _("EXIF marker not found."));
757  return;
758  }
759  d++;
760  ds--;
761  if (ds < 2) {
762  LOG_TOO_SMALL;
763  return;
764  }
765  len = (d[0] << 8) | d[1];
766  exif_log (data->priv->log, EXIF_LOG_CODE_DEBUG, "ExifData",
767  "We have to deal with %i byte(s) of EXIF data.",
768  len);
769  d += 2;
770  ds -= 2;
771  }
772 
773  /*
774  * Verify the exif header
775  * (offset 2, length 6).
776  */
777  if (ds < 6) {
778  LOG_TOO_SMALL;
779  return;
780  }
781  if (memcmp (d, ExifHeader, 6)) {
782  exif_log (data->priv->log, EXIF_LOG_CODE_CORRUPT_DATA,
783  "ExifData", _("EXIF header not found."));
784  return;
785  }
786 
787  exif_log (data->priv->log, EXIF_LOG_CODE_DEBUG, "ExifData",
788  "Found EXIF header.");
789 
790  /* Byte order (offset 6, length 2) */
791  if (ds < 14)
792  return;
793  if (!memcmp (d + 6, "II", 2))
794  data->priv->order = EXIF_BYTE_ORDER_INTEL;
795  else if (!memcmp (d + 6, "MM", 2))
796  data->priv->order = EXIF_BYTE_ORDER_MOTOROLA;
797  else {
798  exif_log (data->priv->log, EXIF_LOG_CODE_CORRUPT_DATA,
799  "ExifData", _("Unknown encoding."));
800  return;
801  }
802 
803  /* Fixed value */
804  if (exif_get_short (d + 8, data->priv->order) != 0x002a)
805  return;
806 
807  /* IFD 0 offset */
808  offset = exif_get_long (d + 10, data->priv->order);
809  exif_log (data->priv->log, EXIF_LOG_CODE_DEBUG, "ExifData",
810  "IFD 0 at %i.", (int) offset);
811 
812  /* Parse the actual exif data (usually offset 14 from start) */
813  exif_data_load_data_content (data, EXIF_IFD_0, d + 6, ds - 6, offset, 0);
814 
815  /* IFD 1 offset */
816  if (offset + 6 + 2 > ds) {
817  return;
818  }
819  n = exif_get_short (d + 6 + offset, data->priv->order);
820  if (offset + 6 + 2 + 12 * n + 4 > ds) {
821  return;
822  }
823  offset = exif_get_long (d + 6 + offset + 2 + 12 * n, data->priv->order);
824  if (offset) {
825  exif_log (data->priv->log, EXIF_LOG_CODE_DEBUG, "ExifData",
826  "IFD 1 at %i.", (int) offset);
827 
828  /* Sanity check. */
829  if (offset > ds - 6) {
830  exif_log (data->priv->log, EXIF_LOG_CODE_CORRUPT_DATA,
831  "ExifData", "Bogus offset.");
832  return;
833  }
834 
835  exif_data_load_data_content (data, EXIF_IFD_1, d + 6, ds - 6, offset, 0);
836  }
837 
838  /*
839  * If we got an EXIF_TAG_MAKER_NOTE, try to interpret it. Some
840  * cameras use pointers in the maker note tag that point to the
841  * space between IFDs. Here is the only place where we have access
842  * to that data.
843  */
844  switch (exif_data_get_type_maker_note (data)) {
845  case EXIF_DATA_TYPE_MAKER_NOTE_OLYMPUS:
846  data->priv->md = exif_mnote_data_olympus_new (data->priv->mem);
847  break;
848  case EXIF_DATA_TYPE_MAKER_NOTE_PENTAX:
849  data->priv->md = exif_mnote_data_pentax_new (data->priv->mem);
850  break;
851  case EXIF_DATA_TYPE_MAKER_NOTE_CANON:
852  data->priv->md = exif_mnote_data_canon_new (data->priv->mem, data->priv->options);
853  break;
854  default:
855  break;
856  }
857 
858  /*
859  * If we are able to interpret the maker note, do so.
860  */
861  if (data->priv->md) {
862  exif_mnote_data_log (data->priv->md, data->priv->log);
863  exif_mnote_data_set_byte_order (data->priv->md,
864  data->priv->order);
865  exif_mnote_data_set_offset (data->priv->md,
866  data->priv->offset_mnote);
867  exif_mnote_data_load (data->priv->md, d, ds);
868  }
869 
870  if (data->priv->options & EXIF_DATA_OPTION_FOLLOW_SPECIFICATION)
871  exif_data_fix (data);
